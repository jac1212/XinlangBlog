        private void initSina()
        {
            for (int x = 0; x < userInfoList.Count; x++)
            {
                if (userInfoList[x]["username"].ToString() == "")
                    break;
                var httpRequest = HttpRequestFactory.CreateHttpRequest(Method.GET) as HttpGet;
                httpRequest.GetRequestToken();
                string url = httpRequest.GetAuthorizationUrl();
                GlobalInfo.requestTokens[x] = httpRequest.Token;
                GlobalInfo.requesTokenSecrets[x] = httpRequest.TokenSecret;
                httpRequest.GetVerifier(userInfoList[x]["username"].ToString(), userInfoList[x]["password"].ToString(), x);
                httpRequest.GetAccessToken();
                GlobalInfo.requestTokens[x] = httpRequest.Token;
                GlobalInfo.requesTokenSecrets[x] = httpRequest.TokenSecret;
            }
            start();
        }
        private void start()
        {
            Thread sendWeiboThread = new Thread(new ThreadStart(sendWeibo));
            sendWeiboThread.Start();
        }
        private void sendWeibo()
        {
            Thread.CurrentThread.IsBackground = true;
            Control.CheckForIllegalCrossThreadCalls = false;
           
            for (int m = 0; m < int.Parse(10+""); m++)
            {
                times++;
                for (int x = 0; x < userInfoList.Count; x++)
                {
                    var httpRequest2 = HttpRequestFactory.CreateHttpRequest(Method.POST);
                    httpRequest2.Token = GlobalInfo.requestTokens[x];
                    httpRequest2.TokenSecret = GlobalInfo.requesTokenSecrets[x];
                    var sendUrl = "http://api.t.sina.com.cn/statuses/update.xml?";
                    httpRequest2.Request(sendUrl, "status=" + HttpUtility.UrlEncode("微博内容" + "  " + times.ToString()));
                    Thread.Sleep(150);
                }
            }
        }
